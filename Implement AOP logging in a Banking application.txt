e, Controller Class
BankController.java
package com.example.bank.controller;

import com.example.bank.service.BankService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/bank")
public class BankController {

    @Autowired
    private BankService bankService;

    @GetMapping("/deposit/{amount}")
    public String deposit(@PathVariable double amount) {
        return bankService.deposit(amount);
    }

    @GetMapping("/withdraw/{amount}")
    public String withdraw(@PathVariable double amount) {
        return bankService.withdraw(amount);
    }

    @GetMapping("/balance")
    public double balance() {
        return bankService.checkBalance();
    }
}

f, AOP Logging Aspect
LoggingAspect.java
package com.example.bank.aspect;

import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.annotation.*;
import org.springframework.stereotype.Component;

@Aspect
@Component
public class LoggingAspect {

    // Before method execution
    @Before("execution(* com.example.bank.service.BankService.*(..))")
    public void logBefore(JoinPoint joinPoint) {

        System.out.println("Method Started: "
                + joinPoint.getSignature().getName());
    }

    // After method execution
    @After("execution(* com.example.bank.service.BankService.*(..))")
    public void logAfter(JoinPoint joinPoint) {

        System.out.println("Method Ended: "
                + joinPoint.getSignature().getName());
    }

    // After returning value
    @AfterReturning(
            pointcut = "execution(* com.example.bank.service.BankService.*(..))",
            returning = "result")
    public void logResult(JoinPoint joinPoint, Object result) {

        System.out.println("Method: "
                + joinPoint.getSignature().getName());

        System.out.println("Returned Value: " + result);
    }
}
